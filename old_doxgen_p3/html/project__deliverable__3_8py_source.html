42 #create the framework and stubs for __readFromCSV, __load, clean, and explore
43 def __readFromCSV(self, filename, header = True):
44 """
45 Reads in the data from a CSV file. The data is stored on a column basis similar to a parquet file to more easily account for data types.
46
47 Args:
48 filename (str): the name of the file to be read in
49 header (bool): whether or not the file has a header
50
51 Returns:
52 data (np.array): the data read in from the CSV file
53 """
54 try:
55 with open(filename, 'r', encoding='utf-8') as f:
56 reader = csv.reader(f)
57 data = list(reader)
58
59 if header:
60 header = data[0]
61 data = data[1:]
62 else:
63 header = [f"col_{i}" for i in range(len(data[0]))]
64
65 #init the dict to store the data
66 columns = {col_name: [] for col_name in header}
67
68 for row in data:
69 for col_name, value in zip(header, row):
70 try:
71 columns[col_name].append(float(value))
72 except:
73 columns[col_name].append(value)
74 #if the value is '' then replace it with np.nan
75 if value == '':
76 columns[col_name][-1] = np.nan
77 else:
78 pass
79
80
81 #ok now convert to numpy array
82 d_type = [(col_name, object if any(isinstance(val, str) for val in columns[col_name]) else float) for col_name in header]
83 self.data = np.array(list(zip(*[columns[col_name] for col_name in header])), dtype = d_type)
84
85 except Exception as e:
86 print(f"Error reading {filename}: {str(e)}")
87
88 return self.data

173 def clean(self, filter_size = (3, 3)):
174 """
175 Cleans the time series data set using a median filter
176
177 Args:
178 filter_size (tuple): the size of the filter to use
179
180 Returns:
181 filtered_data (np.array): the filtered data
182 """
183 print("Cleaning Time Series Data Set...")
184
185 #get the stored data that was read in
186 data = self.datadata
187
188 names = self.datadata.dtype.names
189
190 arr = []
191
192 for row in data:
193 row_arr = []
194 for name in names:
195 row_arr.append(row[name])
196
197 arr.append(row_arr)
198 arr = np.array(arr)
199
200 pad = (filter_size[0] // 2, filter_size[1] // 2)
201 print(f"padding: {pad}")
202 rows, cols = arr.shape
203 print(f"rows: {rows}, cols: {cols}")
204
205 #to store the filtered data
206 filtered_data = np.zeros((rows, cols))
207
208 for i in range(rows):
209 start, end = i - pad[0], i + pad[0] + 1
210
211 #apply median filter over the row
212 filtered_data[i] = np.median(arr[max(start, 0):min(end, rows)], axis = 0)
213
214 self.datadata = filtered_data

There are a lot of nan values towards the end of the data, I'm not sure if this was how it was intended in the prompt, but I still consider nan a value. 762 """
763 Trains the simple KNN classifier
764
765 Args:
766 train_data (np.array): the training data
767 train_labels (np.array): the training labels

769 print("Training simple KNN...")
770
771 self.data = train_data # S(n) T(1)
772 self.labels = train_labels # S(n) T(1)

774 def test(self, test_data, k):
775 """
776 Tests the simple KNN classifier
777 """
778 preds = [] # S(n) T(1)
779
780 for test in test_data:
781 #get the distance from the test to each sample
782 samples = np.linalg.norm(self.data - test, axis=1) # S(n) T(n)
783 #sort the samples and get indices of the k closest
784 sorted_samples = np.argsort(samples)[:k] #S(n log n) T(n log n)
785
786 counts = {} # S(c) T(1) c is the number of classes
787 #count the number of times each label appears
788 for i in sorted_samples:
789 label = self.labels[i] #S(n) T(n)
790 if label in counts:
791 counts[label] += 1 #S(n) T(n)
792 else:
793 counts[label] = 1 #S(n) T(n)
794
795 #get the most common label
796 mode = max(counts, key = counts.get) #S(n) T(n)
797 preds.append(mode) #S(n) T(n)
798
799 self.pred_labels = np.array(preds) #S(n) T(1)
800 return preds
