42 #create the framework and stubs for __readFromCSV, __load, clean, and explore
43 def __readFromCSV(self, filename, header = True):
44 """
45 Reads in the data from a CSV file. The data is stored on a column basis similar to a parquet file to more easily account for data types.
46
47 Args:
48 filename (str): the name of the file to be read in
49 header (bool): whether or not the file has a header
50
51 Returns:
52 data (np.array): the data read in from the CSV file
53 """
54 try:
55 with open(filename, 'r', encoding='utf-8') as f:
56 reader = csv.reader(f)
57 data = list(reader)
58
59 if header:
60 header = data[0]
61 data = data[1:]
62 else:
63 header = [f"col_{i}" for i in range(len(data[0]))]
64
65 #init the dict to store the data
66 columns = {col_name: [] for col_name in header}
67
68 for row in data:
69 for col_name, value in zip(header, row):
70 try:
71 columns[col_name].append(float(value))
72 except:
73 columns[col_name].append(value)
74 #if the value is '' then replace it with np.nan
75 if value == '':
76 columns[col_name][-1] = np.nan
77 else:
78 pass
79
80
81 #ok now convert to numpy array
82 d_type = [(col_name, object if any(isinstance(val, str) for val in columns[col_name]) else float) for col_name in header]
83 self.data = np.array(list(zip(*[columns[col_name] for col_name in header])), dtype = d_type)
84
85 except Exception as e:
86 print(f"Error reading {filename}: {str(e)}")
87
88 return self.data

177 def clean(self, filter_size = (3, 3)):
178 """
179 Cleans the time series data set using a median filter
180
181 Args:
182 filter_size (tuple): the size of the filter to use
183
184 Returns:
185 filtered_data (np.array): the filtered data
186 """
187 print("Cleaning Time Series Data Set...")
188
189 #get the stored data that was read in
190 data = self.datadata
191
192 names = self.datadata.dtype.names
193
194 arr = []
195
196 for row in data:
197 row_arr = []
198 for name in names:
199 row_arr.append(row[name])
200
201 arr.append(row_arr)
202 arr = np.array(arr)
203
204 pad = (filter_size[0] // 2, filter_size[1] // 2)
205 print(f"padding: {pad}")
206 rows, cols = arr.shape
207 print(f"rows: {rows}, cols: {cols}")
208
209 #to store the filtered data
210 filtered_data = np.zeros((rows, cols))
211
212 for i in range(rows):
213 start, end = i - pad[0], i + pad[0] + 1
214
215 #apply median filter over the row
216 filtered_data[i] = np.median(arr[max(start, 0):min(end, rows)], axis = 0)
217
218 self.datadata = filtered_data

582 Cleans the qualitative data set, replacing missing values in numeric columns with the median and replacing missing values in string columns with the mode. There are a lot of nan values towards the end of the data, I'm not sure if this was how it was intended in the prompt, but I still consider nan a value. So if nan is the most common response empty values will be replaced with nan. Some questions have very few responses.
583
584 Returns:
585 cleaned_data (np.array): the cleaned data
586 """
587 print("Cleaning Qual Data Set...")
588

590 for col_name in self.datadata.dtype.names:

592
593 #if the data is numeric

595 #replace missing values with the median limited to 2 decimal places
596 col_data[np.isnan(col_data)] = np.round(np.nanmedian(col_data), 2)
597 else:
598 #convert any np.nan to 'nan' so mode can be used
599 col_data[col_data != col_data] = 'nan'
600 #replace missing values with the mode
601 mode = max(set(col_data), key = list(col_data).count)
602 col_data[col_data == 'nan'] = mode
603
604 self.datadata[col_name] = col_data
605
606 except Exception as e:
607 print(f"Error cleaning {self.filename}: {str(e)}")
